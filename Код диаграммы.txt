Table ingredient {
  ingredient_id integer [primary key]
  name text [not null, unique]  
}

Table dish {
  dish_id integer [primary key]
  name text [not null, unique]
  description text
}

table dish_ingredient {
  dish_id integer
  ingredient_id integer 
  amount decimal [not null] // + check > 0
  unit text [not null]
  note text
  indexes {
    (dish_id, ingredient_id) [pk]
  }
}

table week {
  week_id integer [primary key]
  active  bool [not null, default: true]
}

table week_dish {
  week_id integer 
  dish_id integer 
  day_of_week_id integer 
  indexes {
    (week_id, dish_id, day_of_week_id) [pk]
  }
} 

table day_of_week {
  day_of_week_id integer [primary key] //1..7
  name text [not null, unique]
}
